Transformations - transform

here
translate - movement 
Move sideways and downwardtransform: translate(20px, 10px);
here
rotate - rotation 
Rotate by 20 degreestransform: rotate(20deg);
here
scale - scaling 
Scale up 1.3 timestransform: scale(1.3);
here
skew - slanting 
Horizontal skewtransform: skewX(-20deg);
here
matrix - plane transformation 
Advanced plane transformationtransform: matrix(1, -0.3, 0, 1, 0, 0);

Transitions - transition

hover
transition - smooth change on hover 
Smooth scaling on hovertransition: transform 0.4s;

/* add separately: */
.element:hover {
	transform: scale(1.4);
}

Animations - animation and @keyframes

here
continuous rotation 
Looped rotation animationanimation: spin 2s linear infinite;

@keyframes spin {
	from { transform: rotate(0deg); }
	to { transform: rotate(360deg); }
}
here
color change in a loop 
Background color animationanimation: flash 1.5s infinite;

@keyframes flash {
	0%   { background: red; }
	50%  { background: blue; }
	100% { background: red; }
}
